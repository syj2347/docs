    ```cpp
    /*
     * @lc app=leetcode.cn id=1114 lang=cpp
     *
     * [1114] 按序打印

    // @lc code=start
    #include<semaphore.h>
    class Foo {
    public:
        sem_t pf;
        sem_t ps;
        Foo() {
            sem_init(&pf,0,0);
            sem_init(&ps,0,0);
        }
        void first(function<void()> printFirst) {
            printFirst();
            sem_post(&pf);
        }
        void second(function<void()> printSecond) {
            sem_wait(&pf);
            printSecond();
            sem_post(&ps);
        }
        void third(function<void()> printThird) {
            sem_wait(&ps);
            printThird();
        }
    };
    // @lc code=end
    ```

    ```cpp
    ll fastpow(ll a, ll k, ll p)
        {
            ll res = 1;
            while (k)
            {
                if (k & 1)
                    res = res * a % p;
                a = a * a % p;
                k >>= 1;
            }
            return res;
        }
    ```
